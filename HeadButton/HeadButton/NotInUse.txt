		        public string CategoryInt(string category)
        {
            string formattedValue = null;

            if (category == "cat1")
            {
                formattedValue = "1";
            }
            else if (category == "cat2")
            {
                formattedValue = "2";
            }
            else if (category == "cat3")
            {
                formattedValue = "3";
            }
            else if (category == "cat4")
            {
                formattedValue = "4";
            }
            else if (category == "cat5")
            {
                formattedValue = "5";
            }

            return formattedValue;
        }
